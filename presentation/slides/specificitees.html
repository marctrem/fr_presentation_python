Petit exemple plus concret

Filtrons une liste de nombre!

def filtreur(filtre, liste):
    resultat = []
    for element in liste:
        if filtre(element):
            resultat.append(element)
    return resultat

print(filtreur(lambda x: x>=18, [1, 20, 18, 11, 38]))

[20, 18, 38]

Un filtre plus éfficace?

def filtreur(filtre, liste):
    for element in liste:
        if filtre(element):
            yield element
